Diagramatics provides a set of basic building blocks for creating diagrams. These are polygon, curve, empty, and text.
let poly = polygon([V2(0,0), V2(0,10), V2(10,10)]); draw(poly);
let curv = curve([V2(0,0), V2(0,10), V2(10,10)]); draw(curv);
let sq = square(20); let tx = text('hello'); draw(sq, tx);
Diagramatics also provides a set of basic shapes for creating diagrams. These are rectangle, square, regular_polygon, regular_polygon_side, circle, arc, arrow, arrow2, textvar.
let rect = rectangle(20,10); draw(rect);
let sq = square(20); draw(sq);
let poly = regular_polygon(5, 10); draw(poly);
let poly = regular_polygon_side(5, 10); draw(poly);
let circ = circle(10); draw(circ);
let ar = arc(10, Math.PI/3); draw(ar);
let ar = arrow(V2(10,5), 0.5); draw(ar);
let sq = square(20); let tx = textvar('hello'); draw(sq, tx);
