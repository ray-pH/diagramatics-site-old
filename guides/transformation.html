Diagramatics GitHub Docs Editor Examples ⋅
Diagram that has been created can be transformed. Diagramatics provide the following transformation translate, position, rotate, scale, reflect_over_point, reflect_over_line, reflect, vflip, hflip, move_origin, move_origin_text.
let sq = square(10); let sq0 = sq.stroke('grey').strokedasharray([5,5]); let sq1 = sq.translate(V2(5,5)); draw(sq0, sq1);
let sq = square(10); let sq0 = sq.stroke('grey').strokedasharray([5,5]); let sq1 = sq.position(V2(5,5)); draw(sq0, sq1);
let sq = square(10); let sq0 = sq.stroke('grey').strokedasharray([5,5]); let sq1 = sq.rotate(Math.PI/3); draw(sq0, sq1);
let sq = square(10); let sq0 = sq.stroke('grey').strokedasharray([5,5]); let sq1 = sq.rotate(Math.PI/3, V2(-5,-5)); draw(sq0, sq1);
let sq = square(10); let sq0 = sq.stroke('grey').strokedasharray([5,5]); let sq1 = sq.scale(V2(2,0.5)); draw(sq0, sq1);
let p5 = regular_polygon(5, 10); let p5_0 = p5.stroke('grey').strokedasharray([5,5]); let refl_point = V2(10,5); let p5_1 = p5.reflect_over_point(refl_point); let refl_indicator = text('⮾').position(refl_point).fontsize(11); draw(p5_0, p5_1, refl_indicator);
